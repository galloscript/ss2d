 16 /** @constructor */
 17 ss2d.GPUParticleEmitter = function(x, y, w, h, particleSystem)
 18 {
 19 	
 20 	ss2d.Quad.call(this, x, y, w, h);
 21 	this.mPivotX = this.mWidth*0.5;
 22 	this.mPivotY = this.mHeight*0.5;
 23 	
 24 	if(RENDER_CONTEXT != 'webgl') 
 25 		throw 'ss2d.GPUParticleEmitter requires WebGL Rendering Context';
 26 		
 27 	if(!particleSystem)
 28 		throw 'No particle system provided';
 29 	
 30 	this.mParticleSystem = particleSystem;
 31 	
 32 	if(typeof particleSystem == 'string')
 33 		this.mParticleSystem = ss2d.ResourceManager.loadParticleSystem(particleSystem, this.setup, this);
 34 		
 35 	this.mShowHitBox = false;
 36 };
 37 
 38 goog.inherits(ss2d.GPUParticleEmitter, ss2d.Quad);

 40 ss2d.GPUParticleEmitter.MAX_UPDATE_RATE = 100;

 42 ss2d.GPUParticleEmitter.prototype.setup = function(particleSystem)
 43 {
 44 	this.mParticleSystem = particleSystem;
 45 	this.mTexture = this.mParticleSystem.mTexture;
 46 	
 47 	var desc = this.mParticleSystem.mSystemDescriptor;
 48 	
 49 	this.mSourcePosition = new ss2d.Point(parseFloat(desc['sourcePosition']['-x']),
 50 										  parseFloat(desc['sourcePosition']['-y']));
 51 	this.mSourcePositionVariance = new ss2d.Point(parseFloat(desc['sourcePositionVariance']['-x']),
 52 										  		  parseFloat(desc['sourcePositionVariance']['-y']));		
 53 	this.mGravity = new ss2d.Point(parseFloat(desc['gravity']['-x']),
 54 								   parseFloat(desc['gravity']['-y']));
 55 								   
 56 	this.mAngle = parseFloat(desc['angle']['-value']); 
 57 	this.mAngleVariance = parseFloat(desc['angleVariance']['-value']);								
 58 	this.mSpeed = parseFloat(desc['speed']['-value']); 
 59 	this.mSpeedVariance = parseFloat(desc['speedVariance']['-value']);	
 60     this.mRadialAcceleration = parseFloat(desc['radialAcceleration']['-value']); 
 61     this.mTangentialAcceleration = parseFloat(desc['tangentialAcceleration']['-value']);
 62     this.mRadialAccelVariance = parseFloat(desc['radialAccelVariance']['-value']); 
 63     this.mTangentialAccelVariance = parseFloat(desc['tangentialAccelVariance']['-value']);
 64 	this.mParticleLifeSpan = parseFloat(desc['particleLifeSpan']['-value']); 
 65 	this.mParticleLifespanVariance = parseFloat(desc['particleLifespanVariance']['-value']);			
 66 	this.mStartParticleSize = parseFloat(desc['startParticleSize']['-value']); 
 67 	this.mStartParticleSizeVariance = parseFloat(desc['startParticleSizeVariance']['-value']);
 68 	this.mFinishParticleSize = parseFloat(desc['finishParticleSize']['-value']); 
 69 	var fpsvk = desc['finishParticleSizeVariance']||desc['FinishParticleSizeVariance'];
 70 	this.mFinishParticleSizeVariance = parseFloat(fpsvk['-value']);
 71 	this.mMaxParticles = parseFloat(desc['maxParticles']['-value']);
 72 	this.mDuration = parseFloat(desc['duration']['-value']);
 73     this.mRotationStart = parseFloat(desc['rotationStart']['-value']); 
 74     this.mRotationStartVariance = parseFloat(desc['rotationStartVariance']['-value']);
 75     this.mRotationEnd = parseFloat(desc['rotationEnd']['-value']); 
 76     this.mRotationEndVariance = parseFloat(desc['rotationEndVariance']['-value']);
 77     this.mBlendFuncSource = parseFloat(desc['blendFuncSource']['-value']);
 78     this.mBlendFuncDestination = parseFloat(desc['blendFuncDestination']['-value']);
 79     
 80     this.mStartColor = [parseFloat(desc['startColor']['-red']), 
 81     					parseFloat(desc['startColor']['-green']),
 82     					parseFloat(desc['startColor']['-blue']),
 83     					parseFloat(desc['startColor']['-alpha'])]; 
 84 	this.mStartColorVariance = [parseFloat(desc['startColorVariance']['-red']), 
 85 		    					parseFloat(desc['startColorVariance']['-green']),
 86 		    					parseFloat(desc['startColorVariance']['-blue']),
 87 		    					parseFloat(desc['startColorVariance']['-alpha'])];					
 88 	this.mFinishColor = [parseFloat(desc['finishColor']['-red']), 
 89     					 parseFloat(desc['finishColor']['-green']),
 90     					 parseFloat(desc['finishColor']['-blue']),
 91     					 parseFloat(desc['finishColor']['-alpha'])];
 92 	this.mFinishColorVariance = [parseFloat(desc['finishColorVariance']['-red']), 
 93 		    					 parseFloat(desc['finishColorVariance']['-green']),
 94 		    					 parseFloat(desc['finishColorVariance']['-blue']),
 95 		    					 parseFloat(desc['finishColorVariance']['-alpha'])];
 96     
 97     //These values are used for the special purpose of creating the spinning portal emitter.
 98     this.mMaxRadius = parseFloat(desc['maxRadius']['-value']);					// Max radius at which particles are drawn when rotating
 99 	this.mMaxRadiusVariance = parseFloat(desc['maxRadiusVariance']['-value']);			// Variance of the maxRadius
100 	this.mMinRadius = parseFloat(desc['minRadius']['-value']);					// Radius from source below which a particle dies
101 	this.mRotatePerSecond = parseFloat(desc['rotatePerSecond']['-value']);				// Numeber of degress to rotate a particle around the source pos per second
102 	this.mRotatePerSecondVariance = parseFloat(desc['rotatePerSecondVariance']['-value']);		// Variance in degrees for rotatePerSecond
103 	this.mRadiusSpeed = 0; 		// The speed at which a particle moves from maxRadius to minRadius
104 	
105 	//control properties
106 	this.mRotation = this.mAngle / (180.0 / Math.PI);
107 	this.mEmitterType = parseFloat(desc['emitterType']['-value']);
108 	this.mParticleCount = 0;
109 	this.mEmissionRate = this.mMaxParticles / this.mParticleLifeSpan;
110 	this.mEmitCounter = 0;	
111 	this.mElapsedTime = 0;
112 	
113 	
114 	this.mUseTexture = true;
115 	this.mParticleIndex = 0;
116 	this.mParticles = [];
117 	var particleDataArray = [];
118 	var auxParticleArray = [];
119 	
120 	var worldLocation = this.localToWorld(this.mLocation);
121 	for(var i = 0; i < this.mMaxParticles; ++i)
122 	{
123 		this.mParticles[i] = this.generateParticle(null, worldLocation);
124 		this.mParticles[i].mTimeToLive = 0;
125 		this.mParticles[i].getArray(auxParticleArray);
126 		particleDataArray = particleDataArray.concat(auxParticleArray);
127 		particleDataArray = particleDataArray.concat(auxParticleArray);
128 		particleDataArray = particleDataArray.concat(auxParticleArray);
129 		particleDataArray = particleDataArray.concat(auxParticleArray);
130 	}
131 	this.mParticleCount = this.mMaxParticles;
132 	var gl = ss2d.CURRENT_VIEW.mContext;
133 	var renderSupport = ss2d.CURRENT_VIEW.mRenderSupport;
134 	
135 	this.mParticleDataBuffer = renderSupport.createBuffer(gl.ARRAY_BUFFER, 
136 														  new Float32Array(particleDataArray), 
137 														  ss2d.Particle.SRUCT_SIZE,  particleDataArray.length / ss2d.Particle.SRUCT_SIZE,
138 														  gl.DYNAMIC_DRAW);
139 	delete particleDataArray;
140 	delete auxParticleArray;
141 	
142 	this.mFreeParticles = [];
143 	this.mReady = true;
144 	this.startEmitter();
145 };

157 ss2d.GPUParticleEmitter.MINUS_ONE_TO_ONE = function()
158 {
159 	return (Math.random() * 2) - 1;
160 }

162 ss2d.GPUParticleEmitter.prototype.generateParticle = function(particle, worldLocation, addedLifeTime)
163 {
164 	if(this.mParticleCount >= this.mMaxParticles)
165 		return;
166 		
167 	particle = particle||new ss2d.Particle();
168 	addedLifeTime = addedLifeTime||0;
169 	particle.mTimeStamp = ss2d.CURRENT_VIEW.mTotalTime;
170 	
171 	var m1to1 = ss2d.GPUParticleEmitter.MINUS_ONE_TO_ONE;
172 	
173 	particle.mPosition.mX = worldLocation.mX + (m1to1() * this.mSourcePositionVariance.mX);
174 	particle.mPosition.mY = worldLocation.mY + (m1to1() * this.mSourcePositionVariance.mY);
175 	particle.mStartPos.mX = worldLocation.mX;
176 	particle.mStartPos.mY = worldLocation.mY;
177 	
178 	var newAngle = this.mRotation + (m1to1() * this.mAngleVariance / (180 / Math.PI));
179 	var vector = new ss2d.Point(Math.cos(newAngle), Math.sin(newAngle));
180 	var vectorSpeed = this.mSpeed + (m1to1() * this.mSpeedVariance);
181 	particle.mDirection = ss2d.Point.scalePoint(vector, vectorSpeed);
182 	particle.mDirection.mY *= -1;
183 	particle.mRadius = this.mMaxRadius + (m1to1() * this.mMaxRadiusVariance);
184 	
185 	particle.mRadiusDelta = (this.mMaxRadius / this.mParticleLifeSpan);// * (1.0 / ss2d.GPUParticleEmitter.MAX_UPDATE_RATE);
186 	particle.mAngle  = this.mRotation + (m1to1() * this.mAngleVariance / (180 / Math.PI));
187 	particle.mDegreesPerSecond = (this.mRotatePerSecond + (m1to1() *  this.mRotatePerSecondVariance)) / (180 / Math.PI);
188 	
189 	particle.mRadialAcceleration = this.mRadialAcceleration;
190 	particle.mTangentialAcceleration = this.mTangentialAcceleration;
191 	
192 	particle.mTimeToLive = Math.max(0, this.mParticleLifeSpan + (this.mParticleLifespanVariance * m1to1())) + addedLifeTime;
193 	
194 	var particleStartSize = this.mStartParticleSize + (this.mStartParticleSizeVariance * m1to1());
195 	var particleFinishSize = this.mFinishParticleSize + (this.mFinishParticleSizeVariance * m1to1());
196 	particle.mParticleSizeDelta = ((particleFinishSize - particleStartSize) / particle.mTimeToLive);// * (1.0 / ss2d.GPUParticleEmitter.MAX_UPDATE_RATE);
197 	particle.mParticleSize = Math.max(0, particleStartSize);
198 	
199 	var startColor = particle.mColor;
200 	startColor[0]  = this.mStartColor[0] + (this.mStartColorVariance[0] * m1to1());
201 	startColor[1]  = this.mStartColor[1] + (this.mStartColorVariance[1] * m1to1());
202 	startColor[2]  = this.mStartColor[2] + (this.mStartColorVariance[2] * m1to1());
203 	startColor[3]  = this.mStartColor[3] + (this.mStartColorVariance[3] * m1to1());
204 	
205 	var endColor = [];
206 	endColor[0]  = this.mFinishColor[0] + (this.mFinishColorVariance[0] * m1to1());
207 	endColor[1]  = this.mFinishColor[1] + (this.mFinishColorVariance[1] * m1to1());
208 	endColor[2]  = this.mFinishColor[2] + (this.mFinishColorVariance[2] * m1to1());
209 	endColor[3]  = this.mFinishColor[3] + (this.mFinishColorVariance[3] * m1to1());
210 	
211 	var deltaColor = particle.mDeltaColor;
212 	deltaColor[0] = ((endColor[0] - startColor[0]) / particle.mTimeToLive);// * (1.0 / ss2d.GPUParticleEmitter.MAX_UPDATE_RATE);
213 	deltaColor[1] = ((endColor[1] - startColor[1]) / particle.mTimeToLive);// * (1.0 / ss2d.GPUParticleEmitter.MAX_UPDATE_RATE);
214 	deltaColor[2] = ((endColor[2] - startColor[2]) / particle.mTimeToLive);// * (1.0 / ss2d.GPUParticleEmitter.MAX_UPDATE_RATE);
215 	deltaColor[3] = ((endColor[3] - startColor[3]) / particle.mTimeToLive);// * (1.0 / ss2d.GPUParticleEmitter.MAX_UPDATE_RATE);
216 	
217 	var startA = this.mRotationStart + (this.mRotationStartVariance * m1to1());
218 	var endA = this.mRotationEnd + (this.mRotationEndVariance * m1to1());
219 	
220 	particle.mRotation = startA;
221 	particle.mRotationDelta = (endA - startA) / particle.mTimeToLive;
222 	
223 	this.mParticleCount++;
224 	
225 	return particle;
226 };

230 	ss2d.GPUParticleEmitter.prototype.render = function(support)
231 	{
232 		if(!this.mReady)
233 			return;
234 			
235 		this.mPivotX = this.mWidth*0.5;
236 		this.mPivotY = this.mHeight*0.5;
237 		
238 		
239 		if(this.mShowHitBox)
240 		{
241 			var prevAlpha = this.mAlpha;
242 			this.mAlpha = 0.5;
243 			ss2d.Quad.prototype.render.call(this, support);
244 			this.mAlpha = prevAlpha;
245 		}
246 		
247 		var gl = support.mContext;
248 		
249 		gl.blendFunc(this.mBlendFuncSource, this.mBlendFuncDestination);
250 
251 		var material = support.mMaterials.mGPUParticle;
252 		material.mActiveTexture = this.mTexture.mTextureId;
253 		material.mParticleDataBuffer = this.mParticleDataBuffer;
254 		
255 		material.mEmitterType = this.mEmitterType;
256 		material.mGravity[0] = this.mGravity.mX;
257 		material.mGravity[1] = this.mGravity.mY;
258 		
259 		material.apply(support);
260 		
261 		gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, support.mBuffers.mParticlesQVI);
262 		gl.drawElements(gl.TRIANGLES, this.mMaxParticles * 6, gl.UNSIGNED_SHORT, 0);
263 		
264 		gl.blendFunc(support.mDefaultBlendSource, support.mDefaultBlendDestination);
265 		
266 		
267 	};

269 	ss2d.GPUParticleEmitter.prototype.tick = function(deltaTime)
270 	{
271 		this.updateParticles(deltaTime);
272 	};
273 	
274 	ss2d.GPUParticleEmitter.prototype.updateParticles = function(deltaTime)
275 	{
276 		if(!this.mReady)
277 			return;
278 		
279 		deltaTime = (deltaTime > 2.0 / ss2d.CURRENT_VIEW.mFrameRate) ? 1.0 / ss2d.CURRENT_VIEW.mFrameRate : deltaTime;
280 		var gl = ss2d.CURRENT_VIEW.mContext;
281 		var renderSupport = ss2d.CURRENT_VIEW.mRenderSupport;
282 		if(this.mActive && this.mEmissionRate)
283 		{
284 			var rate = 1.0/this.mEmissionRate;
285 			this.mEmitCounter += deltaTime;
286 			var auxParticleArray = [];
287 			var worldLocation = this.localToWorld(this.mLocation);
288 			
289 			while(this.mParticleCount < this.mMaxParticles && this.mEmitCounter > rate)
290 			{
291 				this.mEmitCounter -= rate;
292 				
293 				if(this.mFreeParticles.length == 0)
294 					continue;
295 				
296 				
297 				var particleIndex = this.mFreeParticles.pop();
298 				var particle = this.generateParticle(this.mParticles[particleIndex], worldLocation);
299 				auxParticleArray = particle.getArray(auxParticleArray);
300 				
301 				//4 times, one per vertex
302 				auxParticleArray = auxParticleArray.concat(auxParticleArray);
303 				auxParticleArray = auxParticleArray.concat(auxParticleArray);
304 				
305 				gl.bindBuffer(gl.ARRAY_BUFFER, this.mParticleDataBuffer);
306 				gl.bufferSubData(gl.ARRAY_BUFFER, 
307 								 particleIndex * 4 * ss2d.Particle.SRUCT_SIZE * 4,
308 								 new Float32Array(auxParticleArray));
309 			}
310 			
311 			this.mElapsedTime += deltaTime;
312 			if(this.mDuration != -1 && this.mDuration < this.mElapsedTime)
313 				this.stopEmitter();
314 		}
315 		
316 		var lifeToRest = deltaTime;
317 		for(var p_index = 0; p_index < this.mMaxParticles; ++p_index)
318 		{
319 			var particle = this.mParticles[p_index];
320 			particle.mTimeToLive -= lifeToRest;
321 			var timeLapse = ss2d.CURRENT_VIEW.mTotalTime - particle.mTimeStamp;
322 			particle.mRadius -= (particle.mRadiusDelta * deltaTime);
323 
324 			if(particle.mTimeToLive <= 0.0 || (this.mEmitterType == 1.0 && particle.mRadius < this.mMinRadius))
325 			{
326 				this.mParticleCount--;
327 				this.mFreeParticles.push(p_index);
328 			}
329 		}
330 		
331 		//console.log(ss2d.CURRENT_VIEW.mTotalTime - this.mParticles[0].mTimeStamp);
332 	};
