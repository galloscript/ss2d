  3 /**
  4  * @fileoverview A frame for of a ss2d.Reel
  5  * @author David Gallardo Moreno (portalg@gmail.com)
  6  */
  7 
  8 goog.provide('ss2d.ReelFrame');
  9 
 10 
 11 /**
 12  * @constructor
 13  * @param {number} x The X texture coordinate
 14  * @param {number} y The Y texture coordinate
 15  * @param {number} width Width in texture pixels
 16  * @param {number} height Height in texture pixels
 17  * @param {(string|ss2d.Texture)} texture
 18  * @param {number=} ox Horizontal offset (optional)
 19  * @param {number=} oy Vertical offset (optional)
 20  * The offsets are used by the ss2d.SpriteReelPlayer to "center" the position of the figure.
 21  */
 22 ss2d.ReelFrame = function(x, y, width, height, offsetX, offsetY)
 23 {
 24 	this.mX = x;
 25 	this.mY = y;
 26 	this.mWidth = width;
 27 	this.mHeight = height;
 28 	
 29 	this.mOffsetX = offsetX||0;
 30 	this.mOffsetY = offsetY||0;
 31 };
 32 
 33 /**
 34  * @param {Array=} clipArray array where the clip information will be dump
 35  * @return {number[4]} array for clipping the proper frame from the texture
 36  */
 37 ss2d.ReelFrame.prototype.dumpClip = function(clipArray)
 38 {
 39 	clipArray = clipArray || new Array(4);
 40 	
 41 	clipArray[0] = this.mX;
 42 	clipArray[1] = this.mY;
 43 	clipArray[2] = this.mWidth;
 44 	clipArray[3] = this.mHeight;
 45 	
 46 	return clipArray;
 47 };
