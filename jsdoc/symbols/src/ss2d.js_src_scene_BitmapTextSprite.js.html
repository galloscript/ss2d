 12 /**
 13  * @constructor
 14  * @extends {ss2d.Quad}
 15  * @param {number=} x Position in x axis 
 16  * @param {number=} y Position in y axis
 17  * @param {string} textString The text that will be displayed.
 18  * @param {string} bitmapFont Bitmap font name.
 19  * @param {number} fontSize size in pixels of the line-height
 20  * @property {*} inherited @see ss2d.Quad
 21  * @property {string} mTextString 
 22  * @property {number} mFontSize
 23  * @property {string}Â mBitmapFont 
 24  * @property {string} mDisplayChars Number of chars that will be displayed, by default -1 (all), used for text scrolling.
 25  * @property {number[5]} mClip
 26  */
 27 ss2d.BitmapTextSprite = function(x, y, textString, bitmapFont, fontSize)
 28 {
 29 	ss2d.Quad.call(this, x, y, 0, 0);
 30 	
 31 	this.mTextString = textString||'';
 32 	this.mFontSize = fontSize||16;
 33 	this.mDisplayChars = -1;
 34 	this.mBitmapFont = ss2d.ResourceManager.loadBitmapFont(bitmapFont);
 35 	
 36 	
 37 	if(COMPILING_CLIENT || COMPILING_OFFLINE)
 38 	{
 39 		this.mGlyphSprite = new ss2d.Sprite;
 40 		this.mClip = [];
 41 		this.mDisplacement = new ss2d.Point();
 42 		
 43 		//WEBGL
 44 		if(RENDER_CONTEXT == 'webgl')
 45 		{
 46 			this.mGlyphSprite.mParent = this;
 47 		}
 48 	}
 49 };
 50 
 51 goog.inherits(ss2d.BitmapTextSprite, ss2d.Quad);
 52 
 53 ss2d.Object.assignClassId(ss2d.BitmapTextSprite, 1007);

 55 /**
 56  * Sets the number of chars that will be displayed, use with ss2d.Tween to create scroll animation.
 57  * @param {number} displayChars
 58  */
 59 ss2d.BitmapTextSprite.prototype.setDisplayChars = function(displayChars)
 60 {
 61 	this.mDisplayChars = Math.floor(displayChars);
 62 }

 66  * @return {number}
 67  */
 68 ss2d.BitmapTextSprite.prototype.getDisplayChars = function()

 70 	return this.mDisplayChars;

 83 	/** @override */
 84 	ss2d.BitmapTextSprite.prototype.render = function(renderSupport)
 85 	{
 86 		renderSupport.pushTransform(this);
 87 		var ctx = renderSupport.mContext;
 88 		try{
 89 			this.mWidth = 0;
 90 			var xTranslation = 0;
 91 			this.mGlyphSprite.mTexture = this.mBitmapFont.mTexture;
 92 			var s = this.mFontSize /this.mBitmapFont.mMidHeight;
 93 			this.mScaleX = this.mScaleY = s;
 94 			var charsToShow = (this.mDisplayChars > -1)?
 95 							  Math.min(this.mDisplayChars, this.mTextString.length):
 96 							  this.mTextString.length;
 97 			
 98 			for(var c = 0; c < charsToShow; ++c)
 99 			{
100 				var charCode = this.mTextString.charCodeAt(c);
101 				if(charCode == 32)
102 				{ 
103 					//ctx.translate(this.mBitmapFont.mMidWidth*0.5, 0);
104 					xTranslation += this.mBitmapFont.mMidWidth*0.6;
105 					continue;
106 				}
107 
108 				this.mBitmapFont.getGlyphClip(charCode, this.mClip);
109 
110 				//this.mGlyphSprite.mRotation = this.mRotation
111 				this.mGlyphSprite.mClip = this.mClip;
112 				this.mGlyphSprite.mWidth =  this.mClip[2];
113 				this.mGlyphSprite.mHeight =  this.mClip[3];
114 				
115 				this.mDisplacement.mX = xTranslation;
116 				this.mDisplacement.mY = this.mClip[5];
117 				this.displacement(c, charCode, xTranslation, this.mClip[5], this.mDisplacement);
118 				
119 				//this.mGlyphSprite.mScaleX = this.mGlyphSprite.mScaleY = s;
120 				//ctx.scale(s, s);
121 				this.mGlyphSprite.mLocation.mX = this.mDisplacement.mX;
122 				this.mGlyphSprite.mLocation.mY = this.mDisplacement.mY;
123 				//ctx.translate(0, this.mClip[5]);
124 				
125 				
126 				
127 				this.mGlyphSprite.render(renderSupport);
128 				//ctx.rotate(this.mRotation);
129 				
130 				/*ctx.drawImage(this.mBitmapFont.mTexture.mTextureElement,
131 							  this.mClip[0],
132 							  this.mClip[1],
133 							  this.mClip[2],
134 							  this.mClip[3],
135 							  0,
136 							  0,
137 							  this.mClip[2], 
138 							  this.mClip[3]);
139 				ctx.translate(0, -this.mClip[5]);
140 				ctx.translate(this.mClip[4], 0);
141 				ctx.scale(1/s, 1/s);*/
142 				xTranslation += this.mClip[4];
143 				
144 			}
145 			this.mWidth = xTranslation;
146 			this.mHeight = this.mFontSize/this.mScaleY;
147 		} catch (t) { }
148 		renderSupport.popTransform();
149 	};
150 }

152 /**
153  * Apply displacement to each character with this hook.
154  * @param {number} charIndex Index in the string.
155  * @param {number} charCode ASII charCode.
156  * @param {number} x The current X location in local space where the char will be rendered.
157  * @param {number} y The current Y location in local space where the char will be rendered.
158  * @param {ss2d.Point} targetDisplacement Output ss2d.Point with the new location info.
159  */
160 ss2d.BitmapTextSprite.prototype.displacement = function(charIndex, charCode, x, y, targetDisplacement)
161 {
162 	targetDisplacement.mX = x;
163 	targetDisplacement.mY = y;
164 };

172 	/** @override */
173 	ss2d.BitmapTextSprite.convert = function(obj)
174 	{
175 		var objBackup =  ss2d.Object.backupAndDeleteObjectProperties(obj);
176 		
177 		ss2d.BitmapTextSprite.call(obj);
178 		
179 		obj.__proto__ = ss2d.BitmapTextSprite.prototype;
180 		
181 		//now obj is a BitmapTextSprite
182 		obj.restoreSerializedProperties(objBackup);
183 		
184 		return obj;
185 	}
186 	
187 	ss2d.BitmapTextSprite.prototype.restoreSerializedProperties = function(objBackup)
188 	{
189 		ss2d.Quad.prototype.restoreSerializedProperties.call(this, objBackup);
190 		
191 		this.mTextString = objBackup['str'];
192 		this.mFontSize = objBackup['fsize'];
193 		this.mBitmapFont = ss2d.ResourceManager.loadBitmapFont(objBackup['bmfname']);
194 		this.mFontStyle = objBackup['fstyle'];
195 		this.mDisplayChars = objBackup['dc'];
196 	}
197 	
198 	ss2d.BitmapTextSprite.createFromSerializedObject = function(obj)
199 	{
200 		var newObj = new ss2d.BitmapTextSprite(obj['x'], 
201 											   obj['y'], 
202 											   obj['str'], 
203 											   obj['bmfname'],
204 											   obj['fsize']);
205 		newObj.mObjectId = obj['doid'];
206 		return newObj;
207 	}
208 	
209 	ss2d.BitmapTextSprite.prototype.interpolateState = function(prevState, nextState, part)
210 	{
211 		ss2d.Quad.prototype.interpolateState.call(this, prevState, nextState, part, deltaTime);
212 		
213 		this.mTextString = prevState['str'];
214 		this.mFontSize = prevState['fsize'];
215 		
216 		if(!this.mBitmapFont ||this.mBitmapFont.mName != prevState['bmfname'])
217 			this.mBitmapFont = ss2d.ResourceManager.loadBitmapFont(prevState['bmfname']);
218 
219 		this.mDisplayChars = Math.floor(prevState['dc'] + (nextState['dc'] - prevState['dc']) * part);
220 	}
221 }
