  3 /**

  7 
  8 goog.provide('ss2d.AudioLoader');
  9 
 10 goog.require('ss2d.ILoader');
 11 goog.require('ss2d.Texture');
 12 
 13 /**
 14  * @constructor
 15  * @implements ss2d.ILoader
 16  * @static
 17  */
 18 ss2d.AudioLoader = {};
 19 
 20 
 21 /** @type {ss2d.IAudioComponent} */
 22 ss2d.AudioLoader.AUDIO_COMPONENT_CLASS = ss2d.HTML5Audio;
 23 if(typeof webkitAudioContext != 'undefined' || typeof AudioContext != 'undefined')
 24 {
 25 	ss2d.AudioLoader.AUDIO_COMPONENT_CLASS = ss2d.WebAudio;
 26 }
 27 
 28 /**
 29  * The extension used in ResourceManager class.
 30  * @static
 31  * @type {string}
 32  */
 33 ss2d.AudioLoader.RESOURCE_EXTENSION = 'sound';
 34 
 35 /**
 36  * Load the audio file
 37  * @param {string} resourceFileName
 38  * @param {function} resourceLoadedCallback
 39  * @return {ss2d.Texture} The loaded texture
 40  */
 41 ss2d.AudioLoader.loadResource = function(resourceFileName, resourceLoadedCallback)
 42 {
 43 	return new ss2d.AudioLoader.AUDIO_COMPONENT_CLASS(resourceFileName, resourceLoadedCallback);
 44 };
