 11 /**
 12  * @constructor 
 13  * @param {number} x
 14  * @param {number} y
 15  * @param {number} w
 16  * @param {number} h
 17  */
 18 ss2d.Rectangle = function(x,y,w,h)
 19 {
 20 	this.mX = x;
 21 	this.mY = y;
 22 	this.mWidth = w;
 23 	this.mHeight = h;	
 24 }

 26 /**
 27  * @param {number} x
 28  * @param {number} y
 29  * @param {number} w
 30  * @param {number} h
 31  * @return {ss2d.Rectangle} this object
 32  */
 33 ss2d.Rectangle.prototype.setValues = function(x,y,w,h)
 34 {
 35 	this.mX = x;
 36 	this.mY = y;
 37 	this.mWidth = w;
 38 	this.mHeight = h;
 39 	
 40 	return this;
 41 }

 43 /**
 44  * @return {ss2d.Rectangle} a copy of this object
 45  */
 46 ss2d.Rectangle.prototype.clone = function()
 47 {
 48 	return new ss2d.Rectangle(this.mX, this.mY, this.mWidth, this.mHeight);
 49 }

 51 /**
 52  * @return {string}
 53  */
 54 ss2d.Rectangle.prototype.toString = function()
 55 {
 56 	return "("+this.mX+","+this.mY+","+this.mWidth+","+this.mHeight+")";
 57 }

 59 /**
 60  * Determines if a coordinate is within the rectangle.
 61  * @param {number} x
 62  * @param {number} y
 63  * @return {boolean}
 64  */
 65 ss2d.Rectangle.prototype.contains = function(x, y)
 66 {
 67 	return  x >= this.mX && y >= this.mY && x <= this.mX + this.mWidth && y <= this.mY + this.mHeight;
 68 }
 69 
 70 /**
 71  * Determines if a point is within the rectangle.
 72  * @param {ss2d.Point} point
 73  * @return {boolean}
 74  */
 75 ss2d.Rectangle.prototype.containsPoint = function(point)
 76 {
 77 	return this.contains(point.mX, point.mY);	
 78 }

104 /**
105  * If the specified rectangle intersects with the rectangle, returns the area of intersection.
106  * @param {ss2d.Rectangle} otherRectangle
107  * @param {ss2d.Rectangle} targetRectangle Empty rectangle to dump the result
108  * @return {ss2d.Rectangle} Area of intersection
109  */
110 ss2d.Rectangle.prototype.intersectionWithRectangle = function(otherRectangle, targetRectangle)
111 {
112 	if(!otherRectangle)
113 	{
114 		return null;
115 	} 
116 	
117 	if(!targetRectangle)
118 	{
119 		targetRectangle = new ss2d.Rectangle(0.0, 0.0, 0.0, 0.0);
120 	} 
121 
122 	var left   = Math.max(this.mX, otherRectangle.mX);
123 	var right  = Math.min(this.mX + this.mWidth, otherRectangle.mX + otherRectangle.mWidth);
124 	var top    = Math.max(this.mY, otherRectangle.mY);
125 	var bottom = Math.min(this.mY + this.mHeight, otherRectangle.mY + otherRectangle.mHeight);
126     
127 	if (!(left > right || top > bottom))
128 	{
129 		targetRectangle.setValues(left, top, right-left, bottom-top);
130 	}
131 	
132 	return targetRectangle;
133 }

135 /**
136  * Adds two rectangles together to create a new Rectangle object (by filling in the space between the two rectangles).
137  * @param {ss2d.Rectangle} otherRectangle
138  */
139 ss2d.Rectangle.prototype.uniteWithRectangle = function(otherRectangle)
140 {
141 	if(!targetRectangle)
142 	{
143 		targetRectangle = this.clone();
144 	} 
145 	
146 	if(!otherRectangle)
147 	{
148 		if(!targetRectangle){
149 			return targetRectangle;
150 		} else {
151 			return targetRectangle.setValues(this.mX, this.mY, this.mWidth, this.mHeight);
152 		}
153 	} 
154 
155 	var left   = Math.min(this.mX, otherRectangle.mX);
156 	var right  = Math.max(this.mX + this.mWidth, otherRectangle.mX + otherRectangle.mWidth);
157 	var top    = Math.min(this.mY, otherRectangle.mY);
158 	var bottom = Math.max(this.mY + this.mHeight, otherRectangle.mY + otherRectangle.mHeight);
159 	
160 	return targetRectangle.setValues(left, top, right - left, bottom - top);
161 }
