 11 /**
 12  * @param {ss2d.View} view
 13  * @constructor
 14  * @implements {EventHandler} 
 15  */
 16 ss2d.Input = function(view)
 17 {
 18 	this.mView = view;
 19 	var canvas = view.mCanvas;
 20 	canvas.style.outline = 'none';
 21 	this.mView.mHaveFocus = true;
 22 	this.mMobileDevice = false;
 23 	if((navigator.userAgent.match(/Android/i)) || (navigator.userAgent.match(/iPhone/i)) || (navigator.userAgent.match(/iPod/i)) ||  (navigator.userAgent.match(/iPad/i)))
 24 	{
 25 		this.mMobileDevice = true;
 26    	}
 27 
 28    	if(!this.mMobileDevice){
 29    		window.addEventListener('mousedown', ss2d.Input.windowMouseEventHandler, true);
 30    	}
 31    	window.addEventListener('keydown', ss2d.Input.handleEventCaller, true);
 32 	window.addEventListener('keyup', ss2d.Input.handleEventCaller, true);
 33    	
 34 	canvas.addEventListener('mousedown', ss2d.Input.handleEventCaller, true);
 35 	canvas.addEventListener('mouseup', ss2d.Input.handleEventCaller, true);
 36 	canvas.addEventListener('mousemove', ss2d.Input.handleEventCaller, true);
 37 
 38 	canvas.addEventListener("touchstart", ss2d.Input.handleEventCaller, false);
 39 	canvas.addEventListener("touchend", ss2d.Input.handleEventCaller, false);
 40 	canvas.addEventListener("touchcancel", ss2d.Input.handleEventCaller, false);
 41 	canvas.addEventListener("touchleave", ss2d.Input.handleEventCaller, false);
 42 
 43 	this.mPressedKeys = {};
 44 	this.mMouseX = -50;
 45 	this.mMouseY = -50;
 46 	this.mPrevMouseX = 0;
 47 	this.mPrevMouseY = 0;
 48 	this.mMouseDown = false;
 49 	this.mMousePoint = new ss2d.Point(this.mMouseX, this.mMouseY);
 50 	this.mPrevMousePoint = new ss2d.Point(this.mPrevMouseX, this.mPrevMouseY);
 51 	this.mPreviousMouseDown = false;
 52 	this.mClicked = false;
 53 }

 93 ss2d.Input.prototype.tick = function()
 94 {
 95 	this.mPrevMouseX = this.mMouseX;
 96 	this.mPrevMouseY = this.mMouseY;
 97 	this.mPrevMousePoint.mX = this.mPrevMouseX;
 98 	this.mPrevMousePoint.mY = this.mPrevMouseY;
 99 	this.mPreviousMouseDown = this.mMouseDown;
100 	this.mMouseDown = this.mClicked;	
101 }
102 
103 //catch all events
104 ss2d.Input.handleEventCaller = function(e)
105 { 
106 	ss2d.CURRENT_VIEW.mInput.handleEvent(e); 
107 }
108 
109 //catch mobile events
110 ss2d.Input.windowMouseEventHandler = function(e)
111 {
112 	var input = ss2d.CURRENT_VIEW.mInput;
113 	if(input.mView.mHaveFocus)
114 	{
115 		input.onFocusOut();
116 	}
117 }
118 
119 /** @override */
120 ss2d.Input.prototype.handleEvent = function(event)
121 {
122 	var r = true;
123 	switch(event.type){
124 	case 'mousedown': case 'touchstart': r = this.onMouseDown(event); break;
125 	case 'mouseup': case 'touchend': case 'touchcancel': r = this.onMouseUp(event); break;
126 	case 'keydown': r = this.onKeyDown(event); break;
127 	case 'keyup': r = this.onKeyUp(event); break;
128 	case 'mousemove': case 'touchmove': r = this.onMouseMove(event); break;
129 	}
130 
131 	return r;
132 }
133 
134 ss2d.Input.prototype.onMouseDown = function(pe)
135 {	
136 	var event = this.mMobileDevice ? window.event.targetTouches[ 0 ] : pe;
137 	
138 	this.mMouseX = event.offsetX || event.pageX - this.mView.mCanvas.offsetLeft;
139 	this.mMouseY = event.offsetY || event.pageY - this.mView.mCanvas.offsetTop;
140 	this.mMousePoint.mX = this.mMouseX;
141 	this.mMousePoint.mY = this.mMouseY;
142 
143 	if((this.mMobileDevice && event) || !this.mMobileDevice){
144 		this.onFocusIn();
145 		this.mClicked = true;
146 		pe.preventDefault();
147 		pe.stopPropagation();
148 		this.mView.mCanvas.addEventListener("touchmove", ss2d.Input.handleEventCaller, false);
149 	}
150 	
151 	return false;
152 }
153 
154 ss2d.Input.prototype.onMouseUp = function(pe)
155 {
156 	var event = this.mMobileDevice ? window.event.targetTouches[ 0 ] : pe;
157 	if((this.mMobileDevice && !event) || !this.mMobileDevice){
158 		this.mMouseDown = false;
159 		this.mClicked = false;
160 		pe.preventDefault();
161 		pe.stopPropagation();
162 		this.mView.mCanvas.removeEventListener("touchmove", ss2d.Input.handleEventCaller, false);
163 	}
164 	return false;
165 }

167 ss2d.Input.prototype.onKeyDown = function(event)
168 {
169 	if(this.mView.mHaveFocus)
170 	{ 
171 		this.mPressedKeys['_'+parseInt(event.keyCode)] = true;
172 		if(event.keyCode == 37 || event.keyCode == 38 || event.keyCode == 39 || event.keyCode == 40 || event.keyCode == 32)
173 		{
174 			event.preventDefault();	
175 		}
176 		return false;
177 	}
178 	return true;
179 }
180 
181 ss2d.Input.prototype.onKeyUp = function(event)
182 {
183 	if(this.mView.mHaveFocus)
184 	{ 
185 		this.mPressedKeys['_'+parseInt(event.keyCode)] = false;
186 		delete this.mPressedKeys['_'+parseInt(event.keyCode)];
187 		return false;
188 	}
189 	return true;
190 }
191 
192 ss2d.Input.prototype.onFocusIn = function(event)
193 {
194 	this.mView.mHaveFocus = true;
195 }
196 
197 ss2d.Input.prototype.onFocusOut = function(event)
198 {
199 	this.mView.mHaveFocus = false;
200 	this.mClicked = false;
201 	//this.mPressedKeys = [];
202 }
203 
204 ss2d.Input.prototype.onMouseMove = function(pe)
205 {
206 	var event = this.mMobileDevice ? window.event.targetTouches[0] : pe;
207 	if(event){
208 		this.mMouseX = event.offsetX || event.pageX - this.mView.mCanvas.offsetLeft;
209 		this.mMouseY = event.offsetY || event.pageY - this.mView.mCanvas.offsetTop;
210 		this.mMousePoint.mX = this.mMouseX;
211 		this.mMousePoint.mY = this.mMouseY;
212 	}
213 }
214 
215 ss2d.Input.prototype.isKeyPressed = function(keycode)
216 {
217 	return (this.mPressedKeys['_'+keycode] != null)?true:false;
218 };

221 ss2d.Input.prototype.toJSON = function()
222 {
223 	var str = '{';
224 	str += '"keys":'+JSON.stringify(this.mPressedKeys)+',';
225 	str += '"mx":'+this.mMouseX+',';
226 	str += '"my":'+this.mMouseY+',';
227 	str += '"md":'+this.mMouseDown;
228 	return str+'}';
229 }
