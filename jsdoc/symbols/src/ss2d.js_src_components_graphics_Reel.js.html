 12 /**
 13  * @constructor
 14  * @param {number} duration
 15  */
 16 ss2d.Reel = function(name, duration)
 17 {
 18 	/** @type {ss2d.ReelFrame[]} */
 19 	this.mFrames = [];
 20 	this.mName = name;
 21 	this.mDuration = duration; //duration in seconds
 22 };
 23 
 24 
 25 //how many time must remain in each frame
 26 ss2d.Reel.prototype.getTimePerFrame = function()
 27 {
 28 	return (this.mFrames.length)?this.mDuration/this.mFrames.length:0;
 29 };

 32 ss2d.Reel.getFramesFromGrid = function(frameWidth, frameHeight, framesPerRow, frames, resultArray, addLineBreaks)
 33 {
 34 	var resultArray = resultArray||[];
 35 	
 36 	for(var f = 0; f < frames; f++)
 37 	{
 38 		var row = Math.floor(f / framesPerRow);
 39 		var column = f - (row * framesPerRow);
 40 		resultArray.push([column*frameWidth, row*frameHeight, frameWidth, frameHeight, 0, 0]);
 41 	}
 42 	
 43 	var str = JSON.stringify(resultArray);
 44 	
 45 	if(addLineBreaks)
 46 	{
 47 		while(str.indexOf('],[') != -1)
 48 		{
 49 			str = str.replace('],[','],\n[');
 50 		}
 51 	}
 52 	return str;
 53 };
