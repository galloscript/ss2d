 17 /**
 18  * @constructor
 19  * @param {number=} x Position in x axis 
 20  * @param {number=} y Position in y axis
 21  * @param {number=} scale Scale in both x and y axis
 22  * @param {number} rotation
 23  * @param {(number|string|number[3]|ss2d.Color)} color
 24  * @param {number} alpha Transparency
 25  * @property {ss2d.Point} mLocation Object location point
 26  * @property {number} mRotation Rotation in rads
 27  * @property {number} mPivotX Pivot in x axis
 28  * @property {number} mPivotY Pivot in y axis
 29  * @property {number} mScaleX Scale in x axis
 30  * @property {number} mScaleY Scale in y axis
 31  * @property {ss2d.DisplayObjectContainer} mParent Parent in scene tree
 32  * @property {ss2d.Color} mColor Color representation
 33  * @property {number} mAlpha Transparency
 34  */
 35 ss2d.DisplayObject = function(x, y, scale, rotation, color, alpha)
 36 {
 37 	x = x || 0.0;
 38 	y = y || 0.0;
 39 	this.mLocation = new ss2d.Point(x, y);
 40 	//this.mX = x || 0.0;
 41 	//this.mY = y || 0.0;
 42 	this.mRotation = rotation || 0.0;
 43 	this.mPivotX = 0.0;
 44 	this.mPivotY = 0.0;
 45 	this.mScaleX = scale || 1.0;
 46 	this.mScaleY = scale || 1.0;
 47 	this.mParent = null;
 48 	
 49 	if(color instanceof ss2d.Color)
 50 	{
 51 		this.mColor = color;
 52 	}
 53 	else
 54 	{
 55 		color = color||0xffffff;
 56 		this.mColor = new ss2d.Color(color);
 57 	}
 58 	
 59 	this.mAlpha = alpha || 1.0;
 60 	
 61 	
 62 	this.mObjectId = ss2d.Object.OBJECT_COUNT++;
 63 	
 64 	if(COMPILING_SERVER)
 65 	{
 66 		this.mSoundList = [];
 67 	}
 68 	
 69 	this.mInheritColor = true;
 70 	this.mInheritAlpha = true;
 71 };

 75 /**
 76  * @param {ss2d.Matrix3} targetMatrix Matrix where dump the result
 77  * @return {ss2d.Matrix3} the local transformation of the object. 
 78  */
 79 ss2d.DisplayObject.prototype.getTransformationMatrix = function(targetMatrix)
 80 {
 81 	targetMatrix = targetMatrix || new ss2d.Matrix3();
 82 	
 83 	if (this.mPivotX != 0.0 || this.mPivotY != 0.0)				{ targetMatrix.translate(-this.mPivotX,-this.mPivotY); }
 84 	if (this.mScaleX != 1.0 || this.mScaleY != 1.0)				{ targetMatrix.scale(this.mScaleX,this.mScaleY); }
 85 	if (this.mRotation != 0.0)			 						{ targetMatrix.rotate(this.mRotation); }
 86 	if (this.mLocation.mX != 0.0 || this.mLocation.mY != 0.0)	{ targetMatrix.translate(this.mLocation.mX,this.mLocation.mY); }
 87     
 88 	return targetMatrix;
 89 };

 91 /**
 92  * @param {ss2d.Matrix3} targetMatrix Matrix where dump the result
 93  * @param {ss2d.DisplayObject} upToParent The top of world transformation (not included)
 94  * @param {boolean} includeHimself
 95  * @return {ss2d.Matrix3} The world transformation of the object. 
 96  */
 97 ss2d.DisplayObject.prototype.getWorldTransformationMatrix = function(targetMatrix, upToParent, includeHimself)
 98 {
 99 	var parentsStack = [];
100 	var currentParent = this;
101 	upToParent = upToParent||null;
102 	while(currentParent && currentParent.mParent != upToParent)
103 	{
104 		parentsStack.push(currentParent.mParent);
105 		currentParent = currentParent.mParent;
106 	}
107 	
108 	targetMatrix = targetMatrix||new ss2d.Matrix3();
109 	var transMatrix = (includeHimself) ? this.getTransformationMatrix(targetMatrix) : targetMatrix;
110 	var curMatrix = new ss2d.Matrix3();
111 	
112 	for(matIndex in parentsStack)
113 	{
114 		curMatrix.identity();
115 		parentsStack[matIndex].getTransformationMatrix(curMatrix);
116 		transMatrix.concatMatrix(curMatrix);
117 	}
118 	
119 	return transMatrix;
120 };

122 /**
123  * Transform a point from world to local coordinate space
124  * @param {ss2d.Point} point
125  * @return {ss2d.Point} Transformed point
126  */
127 ss2d.DisplayObject.prototype.worldToLocal = function(point)
128 {
129 	var worldMatrix = this.getWorldTransformationMatrix();
130 	var invWorldMatrix = worldMatrix.invert();
131 	var transformedPoint = invWorldMatrix.transformPoint(point);
132 	return transformedPoint;
133 };

138  * @return {ss2d.Point} Transformed point
139  */
140 ss2d.DisplayObject.prototype.localToWorld = function(point)
141 {
142 	var worldMatrix = this.getWorldTransformationMatrix();
143 	var transformedPoint = worldMatrix.transformPoint(point);
144 	return transformedPoint;
145 };

166 /**
167  * Returns true if this object bounds intersects with the other object bounds.
168  * @param {ss2d.DisplayObject} otherObject
169  * @return {boolean}
170  */
171 ss2d.DisplayObject.prototype.collideWith = function(otherObject)
172 { 
173 	var thisBounds = this.getBounds();
174 	var otherBounds = otherObject.getBounds();
175 	
176 	if(thisBounds && otherBounds)
177 	{
178 		return thisBounds.intersectsRectangle(otherBounds);
179 	}
180 	
181 	return false;
182 };

235 if(COMPILING_CLIENT)
236 {
237 	/**
238 	 * Cast a JSON parsed object to a object of this type replacing the original object.
239 	 * This is currently not used by the mutiplayer subsystem.
240 	 * @param {Object} obj The JSON parsed object with 'cid' property.
241 	 * @return {ss2d.DisplayObject} A reference to the converted object.
242 	 */
243 	ss2d.DisplayObject.convert = function(obj)
244 	{
245 		var objBackup =  ss2d.Object.backupAndDeleteObjectProperties(obj);
246 		
247 		ss2d.DisplayObject.call(obj);
248 		
249 		obj.__proto__ = ss2d.DisplayObject.prototype;
250 		
251 		//now obj is a DisplayObject
252 		obj.restoreSerializedProperties(objBackup);
253 		
254 		return obj;
255 	}
256 	
257 	/**
258 	 * Copy all properties from a JSON parsed object to this object.
259 	 * This is currently not used by the mutiplayer subsystem.
260 	 * @param {Object} obj The JSON parsed object with 'cid' property.
261 	 */
262 	ss2d.DisplayObject.prototype.restoreSerializedProperties = function(objBackup)
263 	{
264 		this.CLASS_ID = objBackup['cid'];
265 		this.mLocation.mX = objBackup['x'] || this.mLocation.mX;
266 		this.mLocation.mY = objBackup['y'] || this.mLocation.mY;
267 		this.mRotation = objBackup['r'] || this.mRotation;
268 		this.mPivotX = objBackup['px'] || this.mPivotX;
269 		this.mPivotY = objBackup['py'] || this.mPivotY;
270 		this.mScaleX = objBackup['sx'] || this.mScaleX;
271 		this.mScaleY = objBackup['sy'] || this.mScaleY;
272 		this.mColor = new ss2d.Color(objBackup['c']) || this.mColor;
273 		this.mAlpha = objBackup['a'] || this.mAlpha;
274 	}
275 	
276 	/** 
277 	 * Create a new instance of this type from a JSON parsed object.
278 	 * Used by the multiplayer subsystem to create new objects coming from server.
279 	 * @expose
280 	 * @param {Object} obj The JSON parsed object with 'cid' property.
281 	 * @return {ss2d.DisplayObject}  
282 	 */
283 	ss2d.DisplayObject.createFromSerializedObject = function(obj)
284 	{
285 		var newDo = new ss2d.DisplayObject(obj['x'], obj['y'], obj['s'], obj['r'], obj['c']);
286 		newDo.mObjectId = obj['doid'];
287 		return newDo;
288 	}
289 	
290 	/**
291 	 * Interpolate the object state between a previous and future JSON parsed states.
292 	 * This method also calls the tick method for client side updates.
293 	 * @param {Object} prevState
294 	 * @param {Object} nextState
295 	 * @param {number} part Difference between both states to add to the current state.
296 	 * @param {number} deltaTime
297 	 */
298 	ss2d.DisplayObject.prototype.interpolateState = function(prevState, nextState, part, deltaTime)
299 	{	
300 		this.mLocation.mX = (prevState['x']) ? prevState['x'] + (nextState['x'] - prevState['x']) * part : this.mLocation.mX;
301 		this.mLocation.mY = (prevState['y']) ? prevState['y'] + (nextState['y'] - prevState['y']) * part : this.mLocation.mY;
302 		this.mRotation = (prevState['r']) ? prevState['r'] + (nextState['r'] - prevState['r']) * part : this.mRotation;
303 		this.mPivotX = (prevState['px']) ? prevState['px'] + (nextState['px'] - prevState['px']) * part : this.mPivotX;
304 		this.mPivotY = (prevState['py']) ? prevState['py'] + (nextState['py'] - prevState['py']) * part : this.mPivotY;
305 		this.mScaleX = (prevState['sx']) ? prevState['sx'] + (nextState['sx'] - prevState['sx']) * part : this.mScaleX;
306 		this.mScaleY = (prevState['sy']) ? prevState['sy'] + (nextState['sy'] - prevState['sy']) * part : this.mScaleY;
307 		if(prevState['c'])
308 		{
309 			var ca = ss2d.Color.interpolate(prevState['c'], nextState['c'], part).getRGBArray();
310 			this.mColor.setRGB(ca[0],ca[1],ca[2]);
311 		}
312 		this.mAlpha = (prevState['a']) ? prevState['a'] + (nextState['a'] - prevState['a']) * part : this.mAlpha;
313 		
314 		this.playSoundList(prevState);
315 		
316 		//client can update an object state even in an online game, this type of update is intended for non-gameplay 
317 		//related logic, for example when a player receives from server the state "Walking" and the client uses this 
318 		//information to play a SpriteReel animation that represents an avatar of the player walking.
319 		//tick can be defined only for the server, or for both with a block of code for the server and an other for the client.
320 		if(this.tick)
321 		{
322 			this.tick(deltaTime);
323 		}
324 	};
325 	
326 	/**
327 	 * Plays a list of sounds received from the server
328 	 * @param {Object} prevState
329 	 */
330 	ss2d.DisplayObject.prototype.playSoundList = function(prevState)
331 	{
332 		var soundList = prevState['snds'];
333 		if(soundList != null)
334 		{
335 			for(var soundIndex in soundList)
336 			{
337 				var sound = prevState['snds'][childIndex];
338 				try{
339 					ss2d.ResourceManager.loadSound(sound).play();
340 				}catch(e){}
341 			}
342 			
343 			delete prevState['snds'];
344 		}
345 	};
346 }

350 	/**
351 	 * Converts this object into a JSON complaint string.
352 	 * @return {string} The serialized version of the object
353 	 */
354 	ss2d.DisplayObject.prototype.toJSON = function()
355 	{
356 		return '{'+this.getPropertiesJSON()+'}';

358 	
359 	/**
360 	 * @return A string with all properties serialized in JSON format.
361 	 * <pre>
362 	 * SERIALIZACION SPEC
363 	 * 
364 	 * CLASS_ID : cid
365 	 * mObjectId : doid
366 	 * mLocation.mX : x
367 	 * mLocation.mY : y
368 	 * mRotation : r
369 	 * mPivotX : px
370 	 * mPivotY : py
371 	 * mScaleX : sx
372 	 * mScaleY : sy
373 	 * mColor : c
374 	 * mAlpha : a
375 	 * mSoundList : snds
376 	 * </pre>
377 	 */
378 	ss2d.DisplayObject.prototype.getPropertiesJSON = function()
379 	{
380 		var str = '';
381 		
382 		str += '"cid":'+this.CLASS_ID+',';
383 		str += '"doid":'+this.mObjectId+',';
384 		//str += '"mLocation":{"mX":'+this.mLocation.mX+', "mY":'+this.mLocation.mY+'},';
385 		str += '"x":'+Math.floor(this.mLocation.mX*1000.0)/1000.0+',';
386 		str += '"y":'+Math.floor(this.mLocation.mY*1000.0)/1000.0+',';
387 		str += '"r":'+(Math.floor(this.mRotation*100)/100)+',';
388 		str += '"px":'+this.mPivotX+',';
389 		str += '"py":'+this.mPivotY+',';
390 		str += '"sx":'+this.mScaleX+',';
391 		str += '"sy":'+this.mScaleY+',';
392 		str += '"c":'+JSON.stringify(this.mColor.getRGBArray())+',';
393 		str += '"a":'+this.mAlpha;
394 		str += ',"snds":'+JSON.stringify(this.mSoundList);
395 		
396 		return str;
397 	};
398 }
