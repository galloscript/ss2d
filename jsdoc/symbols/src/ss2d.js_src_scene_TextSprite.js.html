 10 goog.require('ss2d.Quad');

 12 /**
 13  * @constructor
 14  * @extends {ss2d.Quad}
 15  * @param {number=} x Position in x axis 
 16  * @param {number=} y Position in y axis
 17  * @param {string} textString The text that will be displayed.
 18  * @param {(number|string|number[3]|ss2d.Color)} color 
 19  * @param {number} fontSize size in pixels of the line-height
 20  * @param {string} fontStyle [normal | italic | oblique] [bold | weight-value from 100 to 900]
 21  * @property {*} inherited @see ss2d.Quad
 22  * @property {string} mTextString 
 23  * @property {number} mFontSize
 24  * @property {string}Â mFontName 
 25  * @property {string} mFontStyle
 26  * @property {string} mDisplayChars Number of chars that will be displayed, by default -1 (all), used for text scrolling.
 27  */
 28 ss2d.TextSprite = function(x, y, textString, color, fontSize, fontName, fontStyle)
 29 {
 30 	ss2d.Quad.call(this, x, y, 0, 0, color);
 31 	
 32 	this.mTextString = textString||'';
 33 	this.mFontSize = fontSize||16;
 34 	this.mFontName = fontName||'Verdana';
 35 	this.mFontStyle = fontStyle||'normal';
 36 	this.mDisplayChars = -1;
 37 	//WEBGL
 38 	//if(RENDER_CONTEXT == 'webgl')
 39 	//{
 40 		
 41 	
 42 	//}
 43 };
 44 
 45 goog.inherits(ss2d.TextSprite, ss2d.Quad);
 46 
 47 ss2d.Object.assignClassId(ss2d.TextSprite, 1007);
 48 
 49 /**
 50  * Sets the number of chars that will be displayed, use with ss2d.Tween to create scroll animation.
 51  * @param {number} displayChars
 52  */
 53 ss2d.TextSprite.prototype.setDisplayChars = function(displayChars)
 54 {
 55 	this.mDisplayChars = Math.floor(displayChars);
 56 }
 57 
 58 /**
 59  * Gets the number of chars that will be displayed, use with ss2d.Tween to create scroll animation.
 60  * @return {number}
 61  */
 62 ss2d.TextSprite.prototype.getDisplayChars = function()
 63 {
 64 	return this.mDisplayChars;
 65 }

 67 if(COMPILING_CLIENT||COMPILING_OFFLINE)
 68 {
 69 	if(RENDER_CONTEXT == 'webgl')
 70 	{
 71 		ss2d.TextSprite.prototype.render = function(renderSupport)
 72 		{
 73 			
 74 		}
 75 	}
 76 	else
 77 	{
 78 		/** @override */
 79 		ss2d.TextSprite.prototype.render = function(renderSupport)
 80 		{
 81 			renderSupport.pushTransform(this);
 82 			var ctx = renderSupport.mContext;
 83 			ctx.font = this.mFontStyle+' '+this.mFontSize+'px '+this.mFontName;
 84 			ctx.fillStyle = this.mColor.getHexString();
 85 			var textToDisplay = this.mTextString;
 86 			
 87 			if(this.mDisplayChars > -1)
 88 				textToDisplay = this.mTextString.substring(0, this.mDisplayChars);
 89 			
 90 			ctx.fillText(textToDisplay, 0, this.mFontSize);	 
 91 			
 92 			renderSupport.popTransform();
 93 		};
 94 	}
 95 }

101 if(COMPILING_CLIENT)
102 {
103 	/** @override */
104 	ss2d.TextSprite.convert = function(obj)
105 	{
106 		var objBackup =  ss2d.Object.backupAndDeleteObjectProperties(obj);
107 		
108 		ss2d.TextSprite.call(obj);
109 		
110 		obj.__proto__ = ss2d.TextSprite.prototype;
111 		
112 		//now obj is a TextSprite
113 		obj.restoreSerializedProperties(objBackup);
114 		
115 		return obj;
116 	}
117 	
118 	ss2d.TextSprite.prototype.restoreSerializedProperties = function(objBackup)
119 	{
120 		ss2d.Quad.prototype.restoreSerializedProperties.call(this, objBackup);
121 		
122 		this.mTextString = objBackup['str'];
123 		this.mFontSize = objBackup['fsize'];
124 		this.mFontName = objBackup['fname'];
125 		this.mFontStyle = objBackup['fstyle'];
126 		this.mDisplayChars = objBackup['dc'];
127 	}
128 	
129 	ss2d.TextSprite.createFromSerializedObject = function(obj)
130 	{
131 		var newObj = new ss2d.TextSprite(obj['x'], 
132 										 obj['y'], 
133 										 obj['str'], 
134 										 obj['c'],  
135 										 obj['fsize'],  
136 										 obj['fname'],  
137 										 obj['fstyle']);
138 		newObj.mObjectId = obj['doid'];
139 		return newObj;
140 	}
141 	
142 	ss2d.TextSprite.prototype.interpolateState = function(prevState, nextState, part)
143 	{
144 		ss2d.Quad.prototype.interpolateState.call(this, prevState, nextState, part, deltaTime);
145 		
146 		this.mTextString = prevState['str'];
147 		this.mFontSize = prevState['fsize'];
148 		this.mFontName = prevState['fname'];
149 		this.mFontStyle = prevState['fstyle'];
150 		this.mDisplayChars = Math.floor(prevState['dc'] + (nextState['dc'] - prevState['dc']) * part);
151 	}
152 }
