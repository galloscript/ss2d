 10 /**
 11  * @constructor
 12  * @param {(ss2d.Color|number|string|Array.<number>)} color
 13  */
 14 ss2d.Color = function(color)
 15 {
 16 	this.setValue(color);
 17 };
 18 
 19 /**
 20  * Sets the color value from various types of color representation formats
 21  * @param {(ss2d.Color|number|string|Array.<number>)} color
 22  */
 23 ss2d.Color.prototype.setValue = function(color)
 24 {
 25 	if(color instanceof ss2d.Color)
 26 	{
 27 		this.mColorArray = color.mColorArray.slice(0);
 28 	}
 29 	else if(color instanceof Array && color.length > 2)
 30 	{
 31 		//color = (color[0] << 16) + (color[1] << 8) + color[2];
 32 		this.mColorArray = color.slice(0);
 33 	}
 34 	else if(typeof color == 'string' || typeof color == 'number')
 35 	{
 36 		if(typeof color == 'string')
 37 		{
 38 			color = parseInt('0x'+(color.replace('#','').substring(0,6)));
 39 		}
 40 		
 41 		color = color & 0xffffff;
 42 		this.mColorArray = [(color & 0xff0000) >> 16, (color & 0xff00) >> 8 , color & 0xff];
 43 	}
 44 };
 45 
 46 /**
 47  * Sets the byte value of red, green and blue color components with values from 0 to 255
 48  * @param {number} red
 49  * @param {number} green
 50  * @param {number} blue
 51  */
 52 ss2d.Color.prototype.setRGB = function(red, green, blue)
 53 {
 54 	this.mColorArray[0] = red;
 55 	this.mColorArray[1] = green;
 56 	this.mColorArray[2] = blue;
 57 };
 58 
 59 ss2d.Color.prototype.getRed = function(){ return this.mColorArray[0]; };
 60 ss2d.Color.prototype.setRed = function(r){ this.mColorArray[0] = r; };
 61 
 62 ss2d.Color.prototype.getGreen = function(){ return this.mColorArray[1]; };
 63 ss2d.Color.prototype.setGreen = function(g){ this.mColorArray[1] = g; };
 64 
 65 ss2d.Color.prototype.getBlue = function(){ return this.mColorArray[2]; };
 66 ss2d.Color.prototype.setBlue = function(b){ this.mColorArray[2] = b; };
 67 
 68 /**
 69  * @return {Array.<number>} Array of bytes with values from 0 to 255
 70  */
 71 ss2d.Color.prototype.getRGBArray = function()
 72 {
 73 	return this.mColorArray;
 74 };
 75 
 76 /**
 77  * @return {string} The Hex string CSS color format: #AABBCC
 78  */
 79 ss2d.Color.prototype.getHexString = function()
 80 {
 81 	var r = Math.floor(this.mColorArray[0]).toString(16); r = (r.length == 1)?'0'+r:r;
 82 	var g = Math.floor(this.mColorArray[1]).toString(16); g = (g.length == 1)?'0'+g:g;
 83 	var b = Math.floor(this.mColorArray[2]).toString(16); b = (b.length == 1)?'0'+b:b;
 84 	
 85 	return '#' + r + g + b;
 86 };
 87 
 88 /**
 89  * @param {number=} The value for the alpha component, 1.0 by default.
 90  * @return {string} The rgba string CSS color format: rgba(red,green,blue,alpha);
 91  */
 92 ss2d.Color.prototype.getRGBAString = function(alpha)
 93 {
 94 	alpha = alpha || '1.0';
 95 	return 'rgba(' + this.mColorArray[0] + ',' + this.mColorArray[1] + ',' + this.mColorArray[2] + ',' + alpha+')';
 96 };
 97 
 98 /**
 99  * @param {Array=} target An already created array to store the values
100  * @return {Array.<number>} Array with float color components values from 0 to 1.
101  */
102 ss2d.Color.prototype.getF32Array = function(target)
103 {
104 	target = target || new Array(3);
105 	target[0] = this.mColorArray[0] / 255;
106 	target[1] = this.mColorArray[1] / 255;
107 	target[2] = this.mColorArray[2] / 255;
108 	return target;
109 };
110 
111 /**
112  * @return {number} The numeric value of the color
113  */
114 ss2d.Color.prototype.getF32 = function()
115 {
116 	return (this.mColorArray[0] << 16) + (this.mColorArray[1] << 8) + this.mColorArray[2];
117 };
118 
119 /**
120  * Interpolate between two color values
121  * @param {Array.<number>} initialColor
122  * @param {Array.<number>} finalColor
123  * @param {number} part The fraction of difference value between two colors.
124  */
125 ss2d.Color.interpolate = function(initialColor, finalColor, part)
126 {
127 	var ic = new ss2d.Color(initialColor);
128 	var fc = new ss2d.Color(finalColor);
129 	ic.mColorArray[0] += (fc.mColorArray[0] - ic.mColorArray[0])*part;
130 	ic.mColorArray[1] += (fc.mColorArray[1] - ic.mColorArray[1])*part;
131 	ic.mColorArray[2] += (fc.mColorArray[2] - ic.mColorArray[2])*part;
132 	delete fc;
133 	return ic;
134 }
