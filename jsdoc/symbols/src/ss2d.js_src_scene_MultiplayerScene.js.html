  3 /**
  4  * @fileoverview A subclass of ss2d.DisplayObjectContainer that implements some multiplayer features.
  5  * @author David Gallardo Moreno (portalg@gmail.com)
  6  */
  7 
  8 goog.provide('ss2d.MultiplayerScene');
  9 
 10 goog.require('ss2d.Defines');
 11 goog.require('ss2d.DisplayObjectContainer');
 12 
 13 /** 
 14  * @constructor 
 15  * @extends {ss2d.DisplayObjectContainer}
 16  * @property {*} inherited @see ss2d.DisplayObjectContainer
 17  * */
 18 ss2d.MultiplayerScene = function()
 19 {
 20 	ss2d.DisplayObjectContainer.call(this);
 21 	
 22 	//seted by the view
 23 	this.mParentView = null;
 24 }
 25 
 26 goog.inherits(ss2d.MultiplayerScene, ss2d.DisplayObjectContainer);
 27 
 28 ss2d.Object.assignClassId(ss2d.MultiplayerScene, 1006);
 29 
 30 if(COMPILING_CLIENT)
 31 {
 32 	/**
 33 	 * @override
 34 	 */
 35 	ss2d.MultiplayerScene.prototype.interpolateState = function(prevState, nextState, part, deltaTime, childrenOnly)
 36 	{
 37 		ss2d.DisplayObjectContainer.prototype.interpolateState.call(this, prevState, nextState, part, deltaTime, true)
 38 	}
 39 }
 40 
 41 if(COMPILING_SERVER)
 42 {
 43 	
 44 	/** @override */
 45 	ss2d.MultiplayerScene.prototype.tick = function(deltaTime)
 46 	{
 47 		this.tickChildren(deltaTime);
 48 		this.mParentView.mComunication.broadcastSceneToClients(this);
 49 	};
 50 	
 51 	/**
 52 	 * Server side method called every time a user connects.
 53  	 * @param {Object} connection
 54 	 */
 55 	ss2d.MultiplayerScene.prototype.onUserConnect = function(connection){};
 56 	
 57 	/**
 58 	 * Server side method called every time a user disconnects.
 59  	 * @param {Object} connection
 60 	 */
 61 	ss2d.MultiplayerScene.prototype.onUserDisconnect = function(connection){};
 62 	
 63 	/**
 64 	 * Server side method called every time a user change his or her name.
 65  	 * @param {Object} connection
 66 	 */
 67 	ss2d.MultiplayerScene.prototype.onUserChangeName = function(connection){};
 68 	
 69 	/**
 70 	 * Server side method called every time a user sent a message not intepreted by the communication interface.
 71 	 * @param {string} commandName
 72 	 * @param {Object} commandParam
 73 	 * @param {Object} connection
 74 	 */
 75 	ss2d.MultiplayerScene.prototype.onUserMessage = function(commandName, commandParam, connection){};
 76 	
 77 	/**
 78 	 * Return a JSON compliant string representation of the entire scene.
 79 	 * @override
 80 	 */
 81 	ss2d.MultiplayerScene.prototype.getPropertiesJSON = function()
 82 	{
 83 		return ss2d.DisplayObjectContainer.prototype.getPropertiesJSON.call(this, true, true);
 84 	}
 85 }
