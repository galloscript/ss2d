  8 goog.provide('ss2d.materials.Textured');
  9 
 10 goog.require('ss2d.ShaderSource');
 11 goog.require('ss2d.ShaderProgram');
 12 
 13 /** @constructor */
 14 ss2d.materials.Textured = function(support)
 15 {
 16 	var gl = support.mContext;
 17 	var vertexShader = new ss2d.ShaderSource().compileSource(ss2d.materials.Textured.VERTEX_SOURCE, gl.VERTEX_SHADER);
 18 	var fragmentShader = new ss2d.ShaderSource().compileSource(ss2d.materials.Textured.FRAGMENT_SOURCE, gl.FRAGMENT_SHADER);
 19 	
 20 	this.mShaderProgram = new ss2d.ShaderProgram([vertexShader, fragmentShader], 
 21 												 ['uMVPMatrix', 'uTMatrix', 'uSampler', 'uColor'], 
 22 												 ['aVertexPosition', 'aTextureCoord']);
 23 								 
 24 	this.mModelViewMatrix = new ss2d.Matrix3();
 25 	this.mTextureCoordMatrix = new ss2d.Matrix3();
 26 	this.mActiveTexture = 0;
 27 	this.mColor = [];
 28 	this.mVertexPositionBuffer = null;
 29 	this.mTextureCoordBuffer = null;
 30 	
 31 	this.mAuxVec4Array = new Float32Array(4);
 32 	this.mAuxMat3Array = new Float32Array(9);
 33 };
 34 
 35 ss2d.materials.Textured.prototype.apply = function(support)
 36 {
 37 	var gl = support.mContext;
 38 	
 39 	this.mShaderProgram.use(gl);
 40 
 41 	support.activeTextureForSampler(this.mActiveTexture, this.mShaderProgram.mUniforms['uSampler'], gl.TEXTURE0);
 42 	
 43 	//gl.uniformMatrix3fv(this.mShaderProgram.mUniforms['uPMatrix'], false, ss2d.CURRENT_VIEW.mProjection.getMatF32Array());
 44 	gl.uniformMatrix3fv(this.mShaderProgram.mUniforms['uTMatrix'], false, this.mTextureCoordMatrix.getMatF32Array(this.mAuxMat3Array));
 45 	
 46 	//var mvpMatrix = ss2d.CURRENT_VIEW.mProjection.clone().concatMatrix(this.mModelViewMatrix); 
 47 	var mvpMatrix = this.mModelViewMatrix.clone().concatMatrix(ss2d.CURRENT_VIEW.mProjection); 
 48 	gl.uniformMatrix3fv(this.mShaderProgram.mUniforms['uMVPMatrix'], false, mvpMatrix.getMatF32Array(this.mAuxMat3Array));	
 49 	
 50 	gl.uniform4fv(this.mShaderProgram.mUniforms['uColor'], this.mColor);
 51 	
 52 	gl.bindBuffer(gl.ARRAY_BUFFER, this.mVertexPositionBuffer);
 53 	gl.vertexAttribPointer(this.mShaderProgram.mAttributes['aVertexPosition'], this.mVertexPositionBuffer.itemSize, gl.FLOAT, false, 0, 0);
 54 	
 55 	gl.bindBuffer(gl.ARRAY_BUFFER, this.mTextureCoordBuffer);
 56 	gl.vertexAttribPointer(this.mShaderProgram.mAttributes['aTextureCoord'], this.mTextureCoordBuffer.itemSize, gl.FLOAT, false, 0, 0);
 57 	
 58 
 59 };
 60 
 61 ss2d.materials.Textured.VERTEX_SOURCE = ''+
 62 	'attribute vec2 aVertexPosition;'+
 63 	'attribute vec2 aTextureCoord;'+
 64 	'uniform mat3 uMVPMatrix;'+
 65 	'varying vec2 vTextureCoord;'+
 66 	'void main(void){'+
 67 		'vTextureCoord = aTextureCoord;'+
 68 		'gl_Position = vec4((uMVPMatrix * vec3(aVertexPosition, 1.0)).xy, 0.0, 1.0);'+
 69 	'}';
 70 
 71 ss2d.materials.Textured.FRAGMENT_SOURCE = ''+
 72 	'precision mediump float;'+
 73 	'uniform sampler2D uSampler;'+
 74 	'uniform mat3 uTMatrix;'+
 75 	'uniform vec4 uColor;'+
 76 	'varying vec2 vTextureCoord;'+
 77 	'void main(void){'+
 78 		'vec4 color = texture2D(uSampler, (uTMatrix * vec3(vTextureCoord, 1.0)).xy);'+
 79 		'gl_FragData[0] = color * uColor;'+
 80 	'}';
