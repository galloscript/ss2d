  8 goog.provide('ss2d.TextureAtlas');
  9 
 10 goog.require('ss2d.Texture');
 11 
 12 /**
 13  * A texture atlas component for TexturePacker JSON (hash) output files.
 14  * @see http://www.codeandweb.com/texturepacker
 15  * @constructor
 16  * @param {string} atlasJSONFileName The resource name
 17  * @param {function} callbackFunction Called when the textureAtlas is loaded
 18  */
 19 ss2d.TextureAtlas = function(atlasJSONFileName, callbackFunction)
 20 {
 21 	this.mName = atlasJSONFileName;
 22 	this.mCallbackFunction = callbackFunction;
 23 	this.mAtlasDescriptor = null;
 24 	this.mTexture = null; 
 25 	
 26 	var atlasFileRequest = new XMLHttpRequest();
 27 	atlasFileRequest.mAtlas = this;
 28 	atlasFileRequest.open("GET", atlasJSONFileName, true);
 29 	atlasFileRequest.responseType = "text";
 30 	atlasFileRequest.onload = function() 
 31 	{
 32 		this.mAtlas.atlasFileLoaded(this.response);
 33 	};
 34 	
 35 	atlasFileRequest.send();
 36 };
 37 
 38 /**
 39  * @private
 40  * @param {Object} frameName
 41  * @param {Object} targetClipArray
 42  */
 43 ss2d.TextureAtlas.prototype.atlasFileLoaded = function(fileData)
 44 {
 45 	this.mAtlasDescriptor = JSON.parse(fileData);
 46 	this.mTexture = new ss2d.Texture(this.mAtlasDescriptor['meta']['image'], this.mCallbackFunction);
 47 };
 48 
 49 
 50 /** 
 51  * Retrieve the clip frame information for a given frame name.
 52  * @param {string} frameName
 53  * @param {number[4]=} targetClipArray Optional array where the clipping information will be copied
 54  * @return {number[4]} The clip array
 55  */
 56 ss2d.TextureAtlas.prototype.getClipFor = function(frameName, targetClipArray)
 57 {
 58 	targetClipArray = targetClipArray||[];
 59 	
 60 	var frame = this.mAtlasDescriptor['frames'][frameName]['frame'];
 61 	
 62 	if(frame)
 63 	{
 64 		targetClipArray[0] = frame['x'];
 65 		targetClipArray[1] = frame['y'];
 66 		targetClipArray[2] = frame['w'];
 67 		targetClipArray[3] = frame['h'];
 68 	}
 69 	
 70 	return targetClipArray;
 71 };
