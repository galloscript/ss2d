  3 /**
  4  * @fileoverview Client side communication interface that uses HTML5 WebSocket API to connect a server.
  5  * Defines a basic protocol to send commands that set a user name, and the user input. 
  6  * It also processes the commands received from server.
  7  * @author David Gallardo Moreno (portalg@gmail.com)
  8  */
  9 
 10 goog.provide('ss2d.ClientCommunicationInterface');
 11 
 12 goog.require('ss2d.Object');
 13 
 14 /**
 15  * @constructor
 16  * @param {ss2d.ClientView} clientView client main view
 17  * @param {string} host
 18  * @param {number} port
 19  */
 20 ss2d.ClientCommunicationInterface = function(clientView, host, port)
 21 {
 22 	this.mClientView = clientView;
 23 	/** @typeÂ {WebSocket} */
 24 	this.mConnection = new WebSocket('ws://'+host+':'+port+'/');
 25 	this.mConnection.mInterface = this;
 26 	this.mConnected = false;
 27 	
 28 	// Log connection
 29 	this.mConnection.onopen = function () {
 30 		console.log('WebSocket connected');
 31 		this.mInterface.mConnected = true;
 32 		this.mInterface.mClientView.onConnect(this);
 33 	};
 34 	
 35 	// Log errors
 36 	this.mConnection.onerror = function (error) {
 37 		console.log('WebSocket Error ' + error);
 38 		this.mInterface.mConnected = false;
 39 	};
 40 	
 41 	this.mConnection.onclose = function () {
 42 		this.mInterface.mConnected = false;
 43 		this.mInterface.mClientView.onDisconnect(this);
 44 	};
 45 	
 46 	// Log messages from the server
 47 	this.mConnection.onmessage = function (event) {
 48 		//var fr = new FileReader();
 49 		//fr.mInterface = this.mInterface;
 50 		//fr.onloadend = function(){ this.mInterface.processServerCommand(this.result);}
 51 		//fr.readAsBinaryString(event.data);
 52 		this.mInterface.processServerCommand(JSON.parse(event.data), this);
 53 		delete event.data;
 54 	};
 55 };
 56 
 57 /**
 58  * Sends a UNAME command with the user name.
 59  * @param {string} userName
 60  */
 61 ss2d.ClientCommunicationInterface.prototype.packAndSendUserName = function(userName)
 62 {
 63 	if(this.mConnected)
 64 	{
 65 		var command = '["UNAME","'+userName+'"]';
 66 		this.mConnection.send(command);
 67 	}
 68 };
 69 
 70 /**
 71  * Sends a UINPUT command with the user input.
 72  * @param {ss2d.Input} inputObject
 73  */
 74 ss2d.ClientCommunicationInterface.prototype.packAndSendInput = function(inputObject)
 75 {
 76 	if(this.mConnected)
 77 	{
 78 		var command = '["UINPUT",'+inputObject.toJSON()+']';
 79 		this.mConnection.send(command);
 80 	}
 81 };
 82 
 83 /**
 84  * Interprets the server commands
 85  * @param {Object[2]} commandPackage
 86  * @param {WebSocket} connection
 87  */
 88 ss2d.ClientCommunicationInterface.prototype.processServerCommand = function(commandPackage, connection)
 89 {
 90 	var commandName = commandPackage[0];
 91 	var commandParam = commandPackage[1];
 92 	
 93 	switch(commandName)
 94 	{
 95 		case 'SCENE':
 96 			this.mClientView.mSceneQueue.push([new Date().getTime(), commandParam]);
 97 		break;
 98 		case 'REJECTED':
 99 			this.mClientView.onServerMessage(commandName, commandParam, connection);
100 		break;
101 		default:
102 			this.mClientView.onServerMessage(commandName, commandParam, connection);
103 		break;
104 	}
105 };
