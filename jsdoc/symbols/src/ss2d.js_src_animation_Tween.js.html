  3 /**
  4  * @fileoverview This class is used with ss2d.Juggler to create animations, 
  5  * calculate the needed time to interpolate between a value and another 
  6  * in the specified time and update the property value every frame.
  7  * @author David Gallardo Moreno (portalg@gmail.com)
  8  */
  9 
 10 goog.provide('ss2d.Tween');
 11 
 12 goog.require('ss2d.TweenedProperty');
 13 goog.require('ss2d.Transitions');
 14 goog.require('ss2d.IAnimatable');
 15 
 16 /** 
 17  * @constructor
 18  * @implements {ss2d.IAnimatable}
 19  * @param {!ss2d.DisplayObject} target
 20  * @param {number} time
 21  * @param {function(number)=} transitionMethod
 22  * @param {number=} delay Starts the animation when this delay time has passed
 23  * @param {ss2d.Tween.LoopType=} looptype Specifies the loop type.
 24  */
 25 ss2d.Tween = function(target, time, transitionMethod, delay, looptype)
 26 {
 27 	this.mTarget = target;		
 28 	this.mProperties = [];
 29 	this.mTransitionMethod = ss2d.Transitions.linear;
 30 	if(transitionMethod) this.mTransitionMethod = transitionMethod;
 31 	
 32 	this.mTotalTime = time;
 33 	this.mCurrentTime = 0.0;
 34 	this.mDelay = 0;
 35 	if(delay) this.mDelay = delay;
 36 	
 37 	this.mLoopCount = 0;
 38 	this.mLoop = ss2d.Tween.LoopType.NONE;
 39 	if(looptype) this.mLoop = looptype;
 40 };
 41 
 42 /**
 43  * Enumeration for loop types
 44  * <pre>
 45  * NONE: 0
 46  * REPEAT: 1
 47  * REVERSE: 2
 48  * </pre>
 49  * @enum {number}
 50  */
 51 ss2d.Tween.LoopType = {
 52 	NONE: 0,
 53 	REPEAT: 1,
 54 	REVERSE: 2
 55 };
 56 
 57 /**
 58  * Creates and store a ss2d.TweenedProperty object
 59  * @param {function} propertyGetter Called to get the current value of the animated property.
 60  * @param {function} propertySetter Called to set the new value of the animated property.
 61  * @param {number} targetValue
 62  */
 63 ss2d.Tween.prototype.animateProperty = function(propertyGetter, propertySetter, targetValue)
 64 {
 65 	this.mProperties.push(new ss2d.TweenedProperty(this.mTarget, propertyGetter, propertySetter, targetValue));
 66 };
 67 
 68 /**
 69  * Created the needed ss2d.TweenedProperty objects to move a ss2d.DisplayObject to the specified coords.
 70  * @param {number} x coord
 71  * @param {number} y coord
 72  */
 73 ss2d.Tween.prototype.moveTo = function(x, y)
 74 {
 75 	this.animateProperty(this.mTarget.getX, this.mTarget.setX, x);
 76 	this.animateProperty(this.mTarget.getY, this.mTarget.setY, y);
 77 };
 78 
 79 /**
 80  * Created the needed ss2d.TweenedProperty object to animate the alpha property of a ss2d.DisplayObject
 81  * @param {number} x coord
 82  * @param {number} y coord
 83  */
 84 ss2d.Tween.prototype.fadeTo = function(alpha)
 85 {
 86 	this.animateProperty(this.mTarget.getAlpha, this.mTarget.setAlpha, alpha);
 87 };
 88 
 89 /**
 90  * Called every frame by the ss2d.Juggler parent object.
 91  * @param {number} deltaTime The time passed since the last frame call (1.0 / frameRate)
 92  */
 93 ss2d.Tween.prototype.tick = function(deltaTime)
 94 {
 95 	if (deltaTime == 0 || (this.mLoop == ss2d.Tween.LoopType.NONE && this.mCurrentTime >= this.mTotalTime))
 96 		return;
 97 	
 98 	if (this.mCurrentTime >= this.mTotalTime)
 99 	{
100 		this.mCurrentTime = 0.0;	
101 		this.mLoopCount++;
102 	}
103 	
104 	var previousTime = this.mCurrentTime;
105 	var restTime = this.mTotalTime - this.mCurrentTime;
106 	var carryOverTime = deltaTime > restTime ? deltaTime - restTime : 0.0;	
107 	this.mCurrentTime = Math.min(this.mTotalTime, this.mCurrentTime + deltaTime);
108 
109 	if (this.mCurrentTime <= 0) return;
110 	
111 	var ratio = this.mCurrentTime / this.mTotalTime;
112 
113 	var invertTransition = (this.mLoop == ss2d.Tween.LoopType.REVERSE && this.mLoopCount % 2 == 1);
114 	
115 	for (var i = 0; i < this.mProperties.length; ++i)
116 	{		
117 		var prop = this.mProperties[i];
118 		if (previousTime <= 0 && this.mCurrentTime > 0) 
119 			prop.mStartValue = prop.currentValue();
120 
121 		var transitionValue = invertTransition? 1.0 - this.mTransitionMethod(1.0 - ratio) : this.mTransitionMethod(ratio);		
122 		prop.setCurrentValue(prop.mStartValue + prop.delta() * transitionValue);
123 	}
124 	
125 	if (previousTime < this.mTotalTime && this.mCurrentTime >= this.mTotalTime)
126 	{
127 		if (this.mLoop == ss2d.Tween.LoopType.REPEAT)
128 		{
129 			for (var i = 0; i < this.mProperties.length; ++i)
130 			{
131 				var prop = this.mProperties[i];
132 				prop.setCurrentValue(prop.mStartValue);
133 			}
134 		}
135 		else if (this.mLoop == ss2d.Tween.LoopType.REVERSE)
136 		{
137 			for (var i = 0; i < this.mProperties.length; ++i)
138 			{
139 				var prop = this.mProperties[i];
140 				prop.setCurrentValue(prop.mEndValue);
141 				prop.mEndValue = prop.mStartValue;
142 			}
143 		}
144 	}
145 	
146 	this.tick(carryOverTime);
147 };
148 
149 /**
150  * Called by the ss2d.Juggler parent object to check if the object must be removed.
151  * @return {boolean} Return true if the animation is completed.
152  */
153 ss2d.Tween.prototype.isComplete = function()
154 {
155 	return this.mCurrentTime >= this.mTotalTime && this.mLoop == ss2d.Tween.LoopType.NONE;
156 };
157 
158 /**
159  * Changes the delay time value
160  * @param {number} delay
161  */
162 ss2d.Tween.prototype.setDelay = function(delay)
163 {
164 	this.mCurrentTime = this.mCurrentTime + this.mDelay - delay;
165 	this.mDelay = delay;
166 };
