 13 /**
 14  * @constructor
 15  * @param {ss2d.MultiplayerScene=} mainScene
 16  * @param {number=} serverPort
 17  * @param {number=} updateRate Number of world states per second that will be updated and broadcasted to all clients.
 18  */
 19 ss2d.ServerView = function(mainScene, serverPort, updateRate)
 20 {
 21 	this.mRunning = false;
 22 	this.mFrameRate = updateRate || 20.0;
 23 	this.mLastFrameTimestamp = new Date().getTime();
 24 	
 25 	ss2d.CURRENT_VIEW = this;
 26 	
 27 	this.mMainScene = mainScene || new ss2d.DisplayObjectContainer();
 28 	this.mMainScene.mParentView = this;
 29 	
 30 	this.mComunication = new ss2d.ServerCommunicationInterface(this, serverPort);
 31 	
 32 	//sounds that must be send to the client on the next command list, once sent, the sound list is empty
 33 	this.mSoundList = [];
 34 	//this.mDeletedObjectList = [];
 35 	
 36 	this.mPhysicalWorld = ss2d.PhysicalWorld.getWorld();
 37 	
 38 	//plug-in development
 39 	this.mPreTickFunctions = [];
 40 	this.mPostTickFunctions = [];
 41 	this.mPreRenderFunctions = [];
 42 	this.mPostRenderFunctions = [];
 43 	
 44 	this.mTotalTime = 0;
 45 };
 46 
 47 /** @type {function} */
 48 ss2d.ServerView.NEXT_FRAME_CALLER = function(){ ss2d.CURRENT_VIEW.nextFrame() };
 49 
 50 /**
 51  * Called every frame to update the scene elements state and broadcast scene to clients.
 52  */
 53 ss2d.ServerView.prototype.nextFrame = function()
 54 {
 55 	//get current time in miliseconds
 56 	var now = new Date().getTime();
 57 	var timePassed = now - this.mLastFrameTimestamp;
 58 	
 59 	var timePassedInSeconds = timePasse/1000.0;
 60 	
 61 	this.mTotalTime += timePassedInSeconds;
 62 	
 63 	for(var methodIndex in this.mPreTickFunctions)
 64 	{ 
 65 		this.mPreTickFunctions[methodIndex].call(null, timePassedInSeconds); 
 66 	}
 67 	
 68 	//update scene
 69 	this.mMainScene.tick(timePassedInSeconds);
 70 	
 71 	for(var methodIndex in this.mPostTickFunctions)
 72 	{ 
 73 		this.mPostTickFunctions[methodIndex].call(null, timePassedInSeconds); 
 74 	}
 75 	
 76 	//update physics
 77 	var worldUpdates = Math.floor(Math.max(1, timePassed/(1000.0/ss2d.PhysicalWorld.UPDATE_RATE)));
 78 	for(var i = 0; i < worldUpdates; ++i)
 79 	{
 80 		this.mPhysicalWorld.tick((timePassed/worldUpdates)/1000.0);
 81 	}
 82 	
 83 	//calculate the delay time for the nextFrame call based on the 
 84 	//frameRate and time spend in update and render operations.
 85 	this.mLastFrameTimestamp = now;
 86 	var wastedTime = (new Date().getTime() - now) / 1000.0;
 87 	var nextFrameTime = (1.0/this.mFrameRate) - wastedTime;
 88 	
 89 	//if game is running call the next frame with the specified time
 90 	if(this.mRunning)
 91 	{ 
 92 		setTimeout(ss2d.ServerView.NEXT_FRAME_CALLER, nextFrameTime * 1000.0);
 93 	}
 94 };
 95 
 96 /**
 97  * Start the main loop if it's ready.
 98  */
 99 ss2d.ServerView.prototype.startLoop = function()
100 {
101 	if(!this.mRunning && this.mMainScene != null)
102 	{
103 		this.mRunning = true;
104 		this.nextFrame();
105 	}	
106 };
107 
108 /**
109  * Pause the loop
110  */
111 ss2d.ServerView.prototype.stopLoop = function()
112 {
113 	this.mRunning = false;	
114 };
