  4  * @fileoverview An ss2d.Sprite with a frame based animation.

  7 
  8 goog.provide('ss2d.ReelSprite');
  9 
 10 goog.require('ss2d.Sprite');
 11 goog.require('ss2d.ReelFrame');
 12 goog.require('ss2d.ReelSet');
 13 
 14 /** @constructor */
 15 ss2d.ReelSprite = function(x, y, w, h, descriptor)
 16 {
 17 	//not Sprite constructor, we only need methods
 18 	ss2d.Quad.call(this, x, y, w, h);
 19 	
 20 	if(!descriptor)
 21 	{
 22 		throw 'Trying to create a sprite reel without a SpriteReelDescriptor';
 23 	}
 24 	
 25 	//Animation info
 26 	this.mDescriptor = descriptor;
 27 	this.mFirstFrame = this.mDescriptor.mFrames[0];
 28 	this.mLoop = true;
 29 	this.mTimeDilation = 1.0;
 30 	
 31 	//Sprite
 32 	this.mTexture = this.mFirstFrame.mTexture;
 33 	this.mClip = this.mFirstFrame.copyClip(new Array(4));
 34 	this.mOffsetX = 0;
 35 	this.mOffsetY = 0;
 36 				
 37 	//Animation state
 38 	this.mPlaying = true;
 39 	this.mComplete = false;
 40 	this.mFrameCount = 0; 
 41 	this.mElapsedTimeCurrentFrame = 0;
 42 };
 43 
 44 goog.inherits(ss2d.ReelSprite, ss2d.Sprite);
 45 
 46 //static
 47 ss2d.Object.assignClassId(ss2d.ReelSprite, 1005);
 48 
 49 //prototype
 50 ss2d.ReelSprite.prototype.playAnimation = function(descriptor)
 51 {
 52 	this.mDescriptor = descriptor || this.mDescriptor;
 53 	this.mPlaying = true;
 54 	this.mComplete = false;
 55 	this.mFrameCount = 0; 
 56 	this.mElapsedTimeCurrentFrame = 0;
 57 }
 58 
 59 ss2d.ReelSprite.prototype.pauseAnimation = function()
 60 {
 61 	this.mPlaying = false;
 62 }
 63 
 64 if(COMPILING_CLIENT || COMPILING_OFFLINE)
 65 {
 66 	ss2d.ReelSprite.prototype.tick = function(deltaTime)
 67 	{
 68 		if(this.mPlaying)
 69 		{
 70 			var timePerFrame = this.mDescriptor.getTimePerFrame() * this.mTimeDilation;	
 71 			
 72 			this.mElapsedTimeCurrentFrame += deltaTime;
 73 			
 74 			//a frame change
 75 			if(this.mElapsedTimeCurrentFrame >= timePerFrame)
 76 			{
 77 				this.mFrameCount++;
 78 				this.mElapsedTimeCurrentFrame -= timePerFrame;
 79 				
 80 				if(this.mFrameCount >= this.mDescriptor.mFrames.length)
 81 				{
 82 					if(this.mLoop)
 83 					{	
 84 						this.mFrameCount = 0;
 85 					}
 86 					else
 87 					{
 88 						this.mFrameCount--;
 89 						this.mComplete = true;
 90 					}
 91 				}
 92 				
 93 				var currentFrame = this.mDescriptor.mFrames[this.mFrameCount];
 94 				
 95 				currentFrame.dumpClip(this.mClip);
 96 				this.mTexture = currentFrame.mTexture;
 97 				this.mOffsetX = currentFrame.mOffsetX;
 98 				this.mOffsetY = currentFrame.mOffsetY;
 99 			}
100 		}
101 	};
102 }
