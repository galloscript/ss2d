 10 goog.require('ss2d.Sprite');
 11 goog.require('ss2d.ReelSet');

 14 /**
 15  * @constructor
 16  * @param {number} x
 17  * @param {number} y
 18  * @param {number} w boundaries
 19  * @param {number} h boundaries
 20  * @param {string} reelSet
 21  * @param {string} playingReel
 22  */
 23 ss2d.ReelSprite = function(x, y, reelScale, reelSet, playingReel)
 24 {
 25 	ss2d.Sprite.call(this, x, y, 1.0, 1.0);
 26 	
 27 	this.mScaleX = this.mScaleY = reelScale||1.0;
 28 	
 29 	if(!reelSet)
 30 	{
 31 		throw 'Trying to create a sprite reel without specifying a ss2d.ReelSet';
 32 	}
 33 
 34 	//Animation info
 35 	this.mReelSetName = reelSet;
 36 	this.mReelSet = null;
 37 	this.mPlayingReel = null;
 38 	this.mDefaultReel = playingReel||false;	
 39 	this.mLoop = true;
 40 	this.mTimeDilation = 1.0;
 41 	
 42 	//Animation state
 43 	this.mPlaying = false;
 44 	this.mComplete = false;
 45 	this.mFrameCount = 0; 
 46 	this.mElapsedTimeCurrentFrame = 0;
 47 	
 48 	//Sprite
 49 	this.mTexture = null;
 50 	this.mClip = [0, 0, 0, 0];
 51 	this.mOffsetX = 0;
 52 	this.mOffsetY = 0;
 53 	
 54 	//setup
 55 	ss2d.ResourceManager.loadReelSet(reelSet, this.setup, this);
 56 
 57 };
 58 
 59 goog.inherits(ss2d.ReelSprite, ss2d.Sprite);
 60 
 61 //static
 62 ss2d.Object.assignClassId(ss2d.ReelSprite, 1005);

 64 /**
 65  * Prepare teh ReelSprite to play animations once the reelset is loaded. 
 66  */
 67 ss2d.ReelSprite.prototype.setup = function()
 68 {
 69 	this.mReelSet = ss2d.ResourceManager.loadReelSet(this.mReelSetName);
 70 	if(!this.mDefaultReel)
 71 	{
 72 		for(var reelName in this.mReelSet.mReels)
 73 		{
 74 			//first reel
 75 			this.playReel(reelName);
 76 			break;
 77 		}
 78 	} else {
 79 		this.playReel(this.mDefaultReel);
 80 	}
 81 	
 82 	this.mTexture = this.mReelSet.mTexture;
 83 	var firstFrame = this.mPlayingReel.mFrames[0];
 84 	firstFrame.dumpClip(this.mClip);
 85 }
 86 
 87 /**
 88  * @param {string} reelName
 89  */
 90 ss2d.ReelSprite.prototype.playReel = function(reelName)
 91 {
 92 	if(this.mReelSet)
 93 	{
 94 		this.mPlayingReel = (reelName && this.mReelSet.mReels[reelName]) ? this.mReelSet.mReels[reelName] : this.mPlayingReel;
 95 		var firstFrame = this.mPlayingReel.mFrames[0];
 96 		firstFrame.dumpClip(this.mClip);
 97 	}
 98 	
 99 	this.mPlaying = true;
100 	this.mComplete = false;
101 	this.mFrameCount = 0; 
102 	this.mElapsedTimeCurrentFrame = 0;
103 }
104 
105 ss2d.ReelSprite.prototype.pauseReel = function()
106 {
107 	this.mPlaying = false;
108 }
109 
110 ss2d.ReelSprite.prototype.resumeReel = function()
111 {
112 	this.mPlaying = true;
113 }

115 if(COMPILING_CLIENT || COMPILING_OFFLINE)
116 {
117 	ss2d.ReelSprite.prototype.tick = function(deltaTime)
118 	{
119 		this.updateReelAnimation(deltaTime);
120 	};
121 	
122 	ss2d.ReelSprite.prototype.updateReelAnimation = function(deltaTime)
123 	{	
124 		if(this.mPlaying && this.mReelSet)
125 		{
126 			if(!this.mTexture)
127 			{
128 				this.mTexture = this.mReelSet.mTexture;
129 			}
130 			
131 			var timePerFrame = this.mPlayingReel.getTimePerFrame();	
132 			
133 			this.mElapsedTimeCurrentFrame += deltaTime * Math.abs(this.mTimeDilation);
134 			
135 			//a frame change
136 			if(Math.abs(this.mTimeDilation) > 0.01 && this.mElapsedTimeCurrentFrame >= timePerFrame)
137 			{
138 				var framesToAdd = Math.floor(this.mElapsedTimeCurrentFrame / timePerFrame);
139 				if(this.mTimeDilation < 0)
140 					framesToAdd *= -1;
141 				this.mFrameCount += framesToAdd;
142 				this.mElapsedTimeCurrentFrame -= timePerFrame;
143 				
144 				if(this.mTimeDilation > 0 && this.mFrameCount >= this.mPlayingReel.mFrames.length)
145 				{
146 					if(this.mLoop)
147 					{	
148 						this.mFrameCount = 0;
149 						this.mElapsedTimeCurrentFrame = 0;
150 					}
151 					else
152 					{
153 						this.mFrameCount--;
154 						this.mComplete = true;
155 						this.mPlaying = false;
156 					}
157 				}
158 				else if(this.mTimeDilation < 0 && this.mFrameCount < 0)
159 				{
160 					if(this.mLoop)
161 					{	
162 						this.mFrameCount = this.mPlayingReel.mFrames.length - 1;
163 						this.mElapsedTimeCurrentFrame = 0;
164 					}
165 					else
166 					{
167 						this.mFrameCount++;
168 						this.mComplete = true;
169 						this.mPlaying = false;
170 					}
171 				}
172 				
173 				var currentFrame = this.mPlayingReel.mFrames[this.mFrameCount];
174 				if(currentFrame)
175 				{
176 					currentFrame.dumpClip(this.mClip);
177 					this.mOffsetX = currentFrame.mOffsetX;
178 					this.mOffsetY = currentFrame.mOffsetY;
179 					
180 					this.mWidth = this.mClip[2];
181 					this.mHeight = this.mClip[3];
182 				}
183 			}
184 		}
185 	};
186 	
187 	ss2d.ReelSprite.prototype.render = function(renderSupport)
188 	{
189 		if(this.mTexture)
190 		{
191 			this.mLocation.mX += this.mOffsetX;
192 			this.mLocation.mY += this.mOffsetY;
193 			ss2d.Sprite.prototype.render.call(this, renderSupport);
194 			this.mLocation.mX -= this.mOffsetX;
195 			this.mLocation.mY -= this.mOffsetY;
196 		}
197 	};
198 }
